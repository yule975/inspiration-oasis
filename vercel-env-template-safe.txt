# 🚀 Vercel 部署环境变量模板（安全版本）
# 复制以下环境变量到 Vercel 项目设置中

# === 基础配置 ===
NODE_ENV=production
NEXT_PUBLIC_APP_URL=https://your-project.vercel.app

# === Supabase 数据库配置 (必需) ===
NEXT_PUBLIC_SUPABASE_URL=https://your-project-id.supabase.co
NEXT_PUBLIC_SUPABASE_ANON_KEY=your_supabase_anon_key
SUPABASE_SERVICE_ROLE_KEY=your_supabase_service_role_key

# === 数据库连接 ===
DATABASE_URL=postgresql://postgres.your-project:password@aws-0-region.pooler.supabase.com:5432/postgres

# === 阿里云邮件推送 (验证码功能) ===
ALIYUN_ACCESS_KEY_ID=your_aliyun_access_key_id
ALIYUN_ACCESS_KEY_SECRET=your_aliyun_access_key_secret
ALIYUN_FROM_ADDRESS=your-email@yourdomain.com
ALIYUN_FROM_NAME=灵感绿洲

# === JWT 认证配置 ===
JWT_SECRET=your_super_strong_jwt_secret_min_32_characters
JWT_EXPIRES_IN=7d

# === 可选配置 ===
# Vercel Analytics (可选)
NEXT_PUBLIC_VERCEL_ANALYTICS=true

# 文件存储 (可选)
NEXT_PUBLIC_SUPABASE_STORAGE_BUCKET=attachments
MAX_FILE_SIZE=10

# 安全设置 (可选)
CORS_ORIGIN=https://your-project.vercel.app
RATE_LIMIT_MAX=1000
LOG_LEVEL=warn

# ========================================
# 📋 部署说明：
# ========================================
# 1. 将上述环境变量逐一添加到 Vercel 项目设置中
# 2. 替换所有占位符为真实值
# 3. 特别注意：请使用您的真实 Supabase 和阿里云配置
# 4. JWT_SECRET 请生成一个强随机字符串
# 5. 完成配置后点击部署

# ⚠️ 安全提醒：
# - 不要将真实密钥提交到代码仓库
# - Vercel 环境变量是安全加密存储的
# - 定期轮换访问密钥确保安全
